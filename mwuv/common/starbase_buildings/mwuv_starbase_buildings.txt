# MWUV
# Since version Game 3.9 Mod 2
# Allows


corvette_yards = {
	icon = "GFX_spaceport_module_corvette_assembly_yard"
	construction_days = 720

	potential = {
		exists = owner
		owner = {
			has_technology = tech_corvettes
		}
	}

	possible = {
		OR = {
			has_starbase_module = shipyard
			starbase_buildable_is_in_queue_before = {
				first = shipyard
				second = corvette_yards
			}
		}
	}

	resources = {
		category = starbase_buildings
		cost = {
			alloys = 500
		}
		upkeep = {
			energy = 10
		}
	}

	custom_tooltip = sm_corvette_yards_desc_modifier

	ai_weight = {
		weight = 100
		modifier = {
			factor = 0
			count_starbase_modules = {
				type = shipyard
				count < 2
			}
		}
		modifier = {
			factor = 0.5
			owner = {
				any_owned_starbase = {
					NOT = {
						is_same_value = root
					}
					has_starbase_building = corvette_yards
				}
			}
		}
	}
	station_modifier = {
		shipsize_corvette_build_speed_mult = 0.5
		shipsize_offspring_corvette_build_speed_mult = 0.5
	}
}
