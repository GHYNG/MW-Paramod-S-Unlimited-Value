# Titan Build Speed (because why not?)

tech_titan_build_speed = {
	cost = @tier5cost2
	area = engineering
	tier = 5
	category = { voidcraft }
	prerequisites = { "tech_titans" }
	weight = @tier5weight2

	modifier = {
		# custom_tooltip = tech_titan_build_speed_effect
		ship_titan_cost_mult = -0.05
		shipsize_titan_build_speed_mult = 0.25
		# shipsize_offspring_titan_build_speed_mult = 0.25 # there is no such thing?
	}

	weight_modifier = {
		modifier = {
			factor = value:tech_weight_likelihood
			research_leader = {
				area = engineering
				has_trait = "leader_trait_expertise_voidcraft"
			}
		}
	}

	ai_weight = {
		modifier = {
			factor = 1.25
			research_leader = {
				area = engineering
				has_trait = "leader_trait_expertise_voidcraft"
			}
		}
	}
}
