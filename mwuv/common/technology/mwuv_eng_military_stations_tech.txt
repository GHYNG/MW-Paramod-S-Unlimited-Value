# defense hulls

# small defense station buff

@station_small_hp_add = 1500
@station_medium_hp_add = 3000
@station_large_hp_add = 6000

tech_defense_platform_1_hull_1 = {
	cost = @tier2cost2
	area = engineering
	tier = 2
	category = { voidcraft }
	prerequisites = { "tech_starbase_3" "tech_space_defense_station_1" }
	is_rare = no
	weight = @tier2weight2

	weight_modifier = {
		factor = 0.50
		modifier = {
			factor = @ap_technological_ascendancy_rare_tech
			has_ascension_perk = ap_technological_ascendancy
		}
		modifier = {
			factor = @federation_perk_factor
			has_federation = yes
			federation = {
				has_federation_perk = rare_tech_boost
				any_member = { has_technology = tech_defense_platform_1_hull_1 }
			}
		}
		modifier = {
			factor = value:tech_weight_likelihood
			has_valid_civic = civic_eager_explorers
		}
		modifier = {
			factor = 1.5
			is_specialist_subject_type = { TYPE = bulwark }
		}
	}

	modifier = {
		shipsize_military_station_small_hull_mult = 0.1
	}
}
